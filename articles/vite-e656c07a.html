import { createHash } from 'node:crypto'

function sha256(value: string) {
  return createHash('sha256').update(value).digest('hex')
}

function buildFingerprint(noteText: string, classifierVersion: string, templateVersion: string) {
  const sourceHash = sha256(noteText)
  const transformHash = sha256(`${classifierVersion}:${templateVersion}`)
  return {
    sourceHash,
    transformHash,
    effectiveHash: sha256(`${sourceHash}:${transformHash}`),
  }
}

interface NoteRecord {
  id: string
  sourceHash: string
  effectiveHash: string
}

interface BuildState {
  classifierVersion: string
  templateVersion: string
  notes: Record<string, NoteRecord>
}

function shouldForceRebuild(prev: BuildState, nextClassifierVersion: string, nextTemplateVersion: string) {
  return (
    prev.classifierVersion !== nextClassifierVersion ||
    prev.templateVersion !== nextTemplateVersion
  )
}

function diffNotes(prev: BuildState, nextNotes: Array<{ id: string; text: string }>, classifierVersion: string, templateVersion: string) {
  const changed: string[] = []

  for (const note of nextNotes) {
    const fp = buildFingerprint(note.text, classifierVersion, templateVersion)
    const prevRecord = prev.notes[note.id]
    if (!prevRecord || prevRecord.effectiveHash !== fp.effectiveHash) {
      changed.push(note.id)
    }
  }

  return changed
}
