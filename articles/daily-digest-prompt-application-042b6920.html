一、執行摘要

本研究針對 daily-digest-prompt 專案進行系統可觀測性與運維監控深度審計，基於專案內部分析（check-health.ps1、state/、logs/、ARCHITECTURE.md 等）與業界最佳實踐，識別出 5 大優化方向：主動告警機制增強、日誌分析自動化、State 健康度監控、運維儀表板建立、告警規則配置化。專案現有 30 個監控區塊、23M 結構化日誌、Autonomous Harness 自主恢復框架，基礎完善但缺乏實時告警觸發與趨勢視覺化，本指南提供可執行的實作方案。

❌ 日誌分析自動化程度低：23MB 結構化日誌缺少自動化異常偵測工具

❌ State 目錄總體健康度監控不足：已有 loop-state 檢查，但缺少整體膨脹趨勢追蹤

❌ 趨勢視覺化缺失：check-health.ps1 輸出文字，難以快速掌握歷史趨勢

❌ 告警規則配置碎片化：閾值分散在多個 YAML（autonomous-harness.yaml、slo.yaml），缺少統一告警配置檔

#### 1.2 實作告警觸發器

新增 tools/alert_trigger.py：


#!/usr/bin/env python
# -*- coding: utf-8 -*-
"""主動告警觸發器（獨立於 check-health.ps1）"""

import json
import yaml
import subprocess
from datetime import datetime, timedelta
from pathlib import Path

PROJECT_ROOT = Path(__file__).parent.parent
ALERT_RULES = PROJECT_ROOT / "config" / "alert-rules.yaml"
STATE_FILE = PROJECT_ROOT / "state" / "alert-suppression.json"
NTFY_TOPIC = "wangsc2025"

def load_alert_rules():
    with open(ALERT_RULES, 'r', encoding='utf-8') as f:
        return yaml.safe_load(f)

def load_suppression_state():
    if not STATE_FILE.exists():
        return {}
    with open(STATE_FILE, 'r', encoding='utf-8') as f:
        return json.load(f)

def save_suppression_state(state):
    with open(STATE_FILE, 'w', encoding='utf-8') as f:
        json.dump(state, f, ensure_ascii=False, indent=2)

def get_metric_value(metric_name):
    """從系統狀態檔案提取指標值"""
    # Session 成功率
    if metric_name == "session_success_rate":
        health_brief = PROJECT_ROOT / "state" / "health-brief-latest.json"
        if health_brief.exists():
            data = json.loads(health_brief.read_text(encoding='utf-8'))
            sr_text = data.get('success_rate_text', 'N/A')
            if sr_text != 'N/A':
                return float(sr_text.rstrip('%'))
        return None
    
    # Phase 失敗次數（近 7 天）
    elif metric_name == "phase_failure_count_7d":
        failure_stats = PROJECT_ROOT / "state" / "failure-stats.json"
        if failure_stats.exists():
            data = json.loads(failure_stats.read_text(encoding='utf-8'))
            return sum(data.get('by_type', {}).values())
        return 0
    
    # Token 使用率
    elif metric_name == "daily_token_usage_pct":
        token_usage = PROJECT_ROOT / "state" / "token-usage-real.json"
        if token_usage.exists():
            data = json.loads(token_usage.read_text(encoding='utf-8'))
            today_usage = data.get('usage_by_day', {}).get(datetime.now().strftime('%Y-%m-%d'), {}).get('total_tokens', 0)
            budget = 500000  # 從 config/budget.yaml 讀取
            return (today_usage / budget) * 100 if budget > 0 else 0
        return 0
    
    # State 目錄大小（MB）
    elif metric_name == "state_directory_size_mb":
        state_dir = PROJECT_ROOT / "state"
        total_bytes = sum(f.stat().st_size for f in state_dir.rglob('*') if f.is_file())
        return round(total_bytes / (1024 * 1024), 2)
    
    # Logs 目錄大小（MB）
    elif metric_name == "logs_directory_size_mb":
        logs_dir = PROJECT_ROOT / "logs"
        total_bytes = sum(f.stat().st_size for f in logs_dir.rglob('*') if f.is_file())
        return round(total_bytes / (1024 * 1024), 2)
    
    # 自主恢復次數（24 小時）
    elif metric_name == "autonomous_recovery_count_24h":
        recovery_queue = PROJECT_ROOT / "state" / "autonomous-recovery-queue.json"
        if recovery_queue.exists():
            data = json.loads(recovery_queue.read_text(encoding='utf-8'))
            cutoff = datetime.now() - timedelta(hours=24)
            return sum(1 for req in data.get('requests', []) 
                      if datetime.fromisoformat(req.get('created_at', '1970-01-01')) > cutoff)
        return 0
    
    return None

def should_alert(rule, value, suppression_state):
    """判斷是否應該發送告警（考慮冷卻期）"""
    rule_name = rule['name']
    cooldown = rule.get('cooldown_minutes', 60)
    
    last_alert_time = suppression_state.get(rule_name)
    if last_alert_time:
        last_time = datetime.fromisoformat(last_alert_time)
        if datetime.now() - last_time < timedelta(minutes=cooldown):
            return False
    
    # 檢查閾值
    threshold = rule['threshold']
    operator = rule['operator']
    
    if operator == '<' and value < threshold:
        return True
    elif operator == '>' and value > threshold:
        return True
    elif operator == '>=' and value >= threshold:
        return True
    elif operator == '<=' and value <= threshold:
        return True
    elif operator == '==' and value == threshold:
        return True
    
    return False

def send_ntfy(notification, value):
    """發送 ntfy 通知"""
    title = notification['title']
    template = notification['template']
    message = template.format(value=value)
    priority = notification.get('priority', 3)
    tags = notification.get('tags', [])
    
    payload = {
        "topic": NTFY_TOPIC,
        "title": title,
        "message": message,
        "priority": priority,
        "tags": tags
    }
    
    json_file = PROJECT_ROOT / "_temp_alert.json"
    json_file.write_text(json.dumps(payload, ensure_ascii=False), encoding='utf-8')
    
    try:
        subprocess.run([
            "curl", "-s", "-X", "POST",
            "https://ntfy.sh",
            "-H", "Content-Type: application/json; charset=utf-8",
            "-d", f"@{json_file}"
        ], check=True, capture_output=True)
    finally:
        json_file.unlink(missing_ok=True)

def main():
    config = load_alert_rules()
    suppression_state = load_suppression_state()
    
    triggered_alerts = []
    
    for rule in config.get('rules', []):
        metric_value = get_metric_value(rule['metric'])
        if metric_value is None:
            continue
        
        if should_alert(rule, metric_value, suppression_state):
            send_ntfy(rule['notification'], metric_value)
            suppression_state[rule['name']] = datetime.now().isoformat()
            triggered_alerts.append(rule['name'])
    
    # 儲存抑制狀態
    if triggered_alerts:
        save_suppression_state(suppression_state)
        print(f"觸發 {len(triggered_alerts)} 個告警：{', '.join(triggered_alerts)}")
    else:
        print("無告警觸發")

if __name__ == "__main__":
    main()

#!/usr/bin/env python
# -*- coding: utf-8 -*-
"""結構化日誌品質評估工具"""

import json
import jsonlines
from pathlib import Path
from datetime import datetime, timedelta
from collections import Counter

PROJECT_ROOT = Path(__file__).parent.parent
LOGS_DIR = PROJECT_ROOT / "logs" / "structured"

def analyze_log_quality():
    """評估日誌品質（結構化程度、完整性、時效性）"""
    results = {
        "total_entries": 0,
        "by_hook": {},
        "missing_fields": Counter(),
        "error_entries": 0,
        "recent_24h_entries": 0,
        "quality_score": 0.0
    }
    
    cutoff = datetime.now() - timedelta(hours=24)
    required_fields = ["timestamp", "hook_name", "tool_name", "status"]
    
    for jsonl_file in LOGS_DIR.glob("*.jsonl"):
        with jsonlines.open(jsonl_file) as reader:
            for entry in reader:
                results["total_entries"] += 1
                
                # 統計各 Hook
                hook_name = entry.get("hook_name", "unknown")
                results["by_hook"][hook_name] = results["by_hook"].get(hook_name, 0) + 1
                
                # 檢查必填欄位
                for field in required_fields:
                    if field not in entry:
                        results["missing_fields"][field] += 1
                
                # 錯誤條目
                if entry.get("status") == "error":
                    results["error_entries"] += 1
                
                # 時效性
                timestamp_str = entry.get("timestamp")
                if timestamp_str:
                    try:
                        timestamp = datetime.fromisoformat(timestamp_str.replace('Z', '+00:00'))
                        if timestamp > cutoff:
                            results["recent_24h_entries"] += 1
                    except:
                        pass
    
    # 計算品質分數（0-100）
    completeness = 1 - (sum(results["missing_fields"].values()) / (results["total_entries"] * len(required_fields)))
    error_rate = results["error_entries"] / results["total_entries"] if results["total_entries"] > 0 else 0
    recency = results["recent_24h_entries"] / results["total_entries"] if results["total_entries"] > 0 else 0
    
    results["quality_score"] = round((completeness * 0.5 + (1 - error_rate) * 0.3 + recency * 0.2) * 100, 2)
    
    return results

def main():
    results = analyze_log_quality()
    
    print("=== 日誌品質分析報告 ===")
    print(f"總條目數: {results['total_entries']}")
    print(f"近 24 小時條目: {results['recent_24h_entries']}")
    print(f"錯誤條目: {results['error_entries']}")
    print(f"品質分數: {results['quality_score']}/100")
    print("\n各 Hook 統計:")
    for hook, count in sorted(results['by_hook'].items(), key=lambda x: x[1], reverse=True):
        print(f"  {hook}: {count}")
    
    if results['missing_fields']:
        print("\n缺失欄位統計:")
        for field, count in results['missing_fields'].most_common():
            print(f"  {field}: {count} 次")
    
    # 寫入狀態檔
    state_file = PROJECT_ROOT / "state" / "log-quality.json"
    state_file.write_text(json.dumps(results, ensure_ascii=False, indent=2), encoding='utf-8')
    print(f"\n狀態已儲存至 {state_file}")

if __name__ == "__main__":
    main()

#### 2.2 異常模式偵測

新增 tools/log_anomaly_detector.py（基於統計異常偵測）：


#!/usr/bin/env python
# -*- coding: utf-8 -*-
"""日誌異常模式偵測器（Z-score 異常偵測）"""

import json
import jsonlines
import statistics
from pathlib import Path
from datetime import datetime, timedelta
from collections import defaultdict

PROJECT_ROOT = Path(__file__).parent.parent
LOGS_DIR = PROJECT_ROOT / "logs" / "structured"

def detect_anomalies():
    """偵測異常模式（頻率異常、錯誤率飆升、耗時異常）"""
    # 統計過去 7 天的每小時錯誤數
    hourly_errors = defaultdict(int)
    hourly_total = defaultdict(int)
    durations = []
    
    cutoff = datetime.now() - timedelta(days=7)
    
    for jsonl_file in LOGS_DIR.glob("*.jsonl"):
        with jsonlines.open(jsonl_file) as reader:
            for entry in reader:
                timestamp_str = entry.get("timestamp")
                if not timestamp_str:
                    continue
                
                try:
                    timestamp = datetime.fromisoformat(timestamp_str.replace('Z', '+00:00'))
                    if timestamp < cutoff:
                        continue
                    
                    hour_key = timestamp.strftime('%Y-%m-%d-%H')
                    hourly_total[hour_key] += 1
                    
                    if entry.get("status") == "error":
                        hourly_errors[hour_key] += 1
                    
                    duration = entry.get("duration_ms")
                    if duration is not None:
                        durations.append(duration)
                except:
                    pass
    
    # 計算錯誤率異常（Z-score > 2）
    error_rates = [hourly_errors[k] / hourly_total[k] for k in hourly_total if hourly_total[k] > 0]
    anomalies = []
    
    if len(error_rates) > 2:
        mean_error_rate = statistics.mean(error_rates)
        stdev_error_rate = statistics.stdev(error_rates)
        
        for hour, total in hourly_total.items():
            if total == 0:
                continue
            error_rate = hourly_errors[hour] / total
            z_score = (error_rate - mean_error_rate) / stdev_error_rate if stdev_error_rate > 0 else 0
            
            if abs(z_score) > 2:
                anomalies.append({
                    "type": "error_rate_spike",
                    "hour": hour,
                    "error_rate": round(error_rate, 3),
                    "z_score": round(z_score, 2),
                    "description": f"{hour} 錯誤率異常（{error_rate:.1%}，Z-score={z_score:.2f}）"
                })
    
    # 計算耗時異常
    if len(durations) > 10:
        mean_duration = statistics.mean(durations)
        stdev_duration = statistics.stdev(durations)
        threshold = mean_duration + 3 * stdev_duration
        
        outliers = [d for d in durations if d > threshold]
        if outliers:
            anomalies.append({
                "type": "duration_outlier",
                "count": len(outliers),
                "max_duration": max(outliers),
                "threshold": round(threshold, 2),
                "description": f"發現 {len(outliers)} 次異常耗時（>{threshold:.0f}ms）"
            })
    
    return anomalies

def main():
    anomalies = detect_anomalies()
    
    print("=== 日誌異常偵測報告 ===")
    if not anomalies:
        print("未發現異常模式")
    else:
        print(f"發現 {len(anomalies)} 個異常:")
        for a in anomalies:
            print(f"  - {a['description']}")
    
    # 寫入狀態檔
    state_file = PROJECT_ROOT / "state" / "log-anomalies.json"
    state_file.write_text(json.dumps({
        "detected_at": datetime.now().isoformat(),
        "anomalies": anomalies
    }, ensure_ascii=False, indent=2), encoding='utf-8')

if __name__ == "__main__":
    main()

#!/usr/bin/env python
# -*- coding: utf-8 -*-
"""State 目錄健康度監控"""

import json
from pathlib import Path
from datetime import datetime, timedelta

PROJECT_ROOT = Path(__file__).parent.parent
STATE_DIR = PROJECT_ROOT / "state"

def analyze_state_health():
    """分析 state/ 目錄健康度"""
    results = {
        "total_files": 0,
        "total_size_mb": 0.0,
        "by_extension": {},
        "stale_files": [],
        "large_files": [],
        "trend": []
    }
    
    cutoff = datetime.now() - timedelta(days=7)
    
    for file in STATE_DIR.rglob('*'):
        if not file.is_file():
            continue
        
        results["total_files"] += 1
        file_size_mb = file.stat().st_size / (1024 * 1024)
        results["total_size_mb"] += file_size_mb
        
        # 按副檔名統計
        ext = file.suffix or "no_ext"
        results["by_extension"][ext] = results["by_extension"].get(ext, 0) + 1
        
        # 過期檔案（7 天未修改）
        mtime = datetime.fromtimestamp(file.stat().st_mtime)
        if mtime < cutoff:
            results["stale_files"].append({
                "path": str(file.relative_to(PROJECT_ROOT)),
                "size_mb": round(file_size_mb, 3),
                "last_modified": mtime.isoformat()
            })
        
        # 大檔案（>1MB）
        if file_size_mb > 1:
            results["large_files"].append({
                "path": str(file.relative_to(PROJECT_ROOT)),
                "size_mb": round(file_size_mb, 2)
            })
    
    results["total_size_mb"] = round(results["total_size_mb"], 2)
    
    # 載入歷史趨勢（從 state/state-health-trend.json）
    trend_file = STATE_DIR / "state-health-trend.json"
    if trend_file.exists():
        trend_data = json.loads(trend_file.read_text(encoding='utf-8'))
        results["trend"] = trend_data.get("trend", [])
    
    # 新增當前快照到趨勢
    results["trend"].append({
        "date": datetime.now().strftime('%Y-%m-%d'),
        "total_files": results["total_files"],
        "total_size_mb": results["total_size_mb"]
    })
    
    # 僅保留最近 30 天
    results["trend"] = results["trend"][-30:]
    
    return results

def main():
    results = analyze_state_health()
    
    print("=== State 目錄健康度報告 ===")
    print(f"總檔案數: {results['total_files']}")
    print(f"總大小: {results['total_size_mb']} MB")
    print(f"過期檔案（7 天未修改）: {len(results['stale_files'])}")
    print(f"大檔案（>1MB）: {len(results['large_files'])}")
    
    if results['large_files']:
        print("\n大檔案清單:")
        for f in sorted(results['large_files'], key=lambda x: x['size_mb'], reverse=True)[:10]:
            print(f"  {f['path']}: {f['size_mb']} MB")
    
    # 寫入狀態檔與趨勢
    state_file = STATE_DIR / "state-health.json"
    state_file.write_text(json.dumps(results, ensure_ascii=False, indent=2), encoding='utf-8')
    
    trend_file = STATE_DIR / "state-health-trend.json"
    trend_file.write_text(json.dumps({
        "updated_at": datetime.now().isoformat(),
        "trend": results["trend"]
    }, ensure_ascii=False, indent=2), encoding='utf-8')
    
    print(f"\n狀態已儲存至 {state_file}")

if __name__ == "__main__":
    main()

#!/usr/bin/env python
# -*- coding: utf-8 -*-
"""運維儀表板生成器（Markdown 格式）"""

import json
from pathlib import Path
from datetime import datetime

PROJECT_ROOT = Path(__file__).parent.parent

def generate_dashboard():
    """生成運維儀表板 Markdown"""
    
    # 載入各項狀態
    health_brief = PROJECT_ROOT / "state" / "health-brief-latest.json"
    token_usage = PROJECT_ROOT / "state" / "token-usage-real.json"
    state_health = PROJECT_ROOT / "state" / "state-health.json"
    log_quality = PROJECT_ROOT / "state" / "log-quality.json"
    
    dashboard = ["# Daily-Digest-Prompt 運維儀表板\n"]
    dashboard.append(f"**生成時間**：{datetime.now().strftime('%Y-%m-%d %H:%M:%S')}\n\n---\n\n")
    
    # 系統健康度
    dashboard.append("## 系統健康度\n\n")
    if health_brief.exists():
        hb = json.loads(health_brief.read_text(encoding='utf-8'))
        sr = hb.get('success_rate_text', 'N/A')
        issues = hb.get('issue_summary', '無')
        dashboard.append(f"- **Session 成功率**：{sr}\n")
        dashboard.append(f"- **問題**：{issues}\n")
    else:
        dashboard.append("（無健康簡報）\n")
    
    dashboard.append("\n---\n\n")
    
    # Token 使用狀態
    dashboard.append("## Token 使用狀態\n\n")
    if token_usage.exists():
        tu = json.loads(token_usage.read_text(encoding='utf-8'))
        today = datetime.now().strftime('%Y-%m-%d')
        today_usage = tu.get('usage_by_day', {}).get(today, {})
        total_tokens = today_usage.get('total_tokens', 0)
        dashboard.append(f"- **今日用量**：{total_tokens:,} tokens\n")
    else:
        dashboard.append("（無 Token 使用資料）\n")
    
    dashboard.append("\n---\n\n")
    
    # State 目錄健康度
    dashboard.append("## State 目錄健康度\n\n")
    if state_health.exists():
        sh = json.loads(state_health.read_text(encoding='utf-8'))
        dashboard.append(f"- **總檔案數**：{sh.get('total_files', 0)}\n")
        dashboard.append(f"- **總大小**：{sh.get('total_size_mb', 0)} MB\n")
        dashboard.append(f"- **過期檔案**：{len(sh.get('stale_files', []))}\n")
        dashboard.append(f"- **大檔案**：{len(sh.get('large_files', []))}\n")
    else:
        dashboard.append("（無 State 健康度資料）\n")
    
    dashboard.append("\n---\n\n")
    
    # 日誌品質
    dashboard.append("## 日誌品質\n\n")
    if log_quality.exists():
        lq = json.loads(log_quality.read_text(encoding='utf-8'))
        dashboard.append(f"- **總條目數**：{lq.get('total_entries', 0)}\n")
        dashboard.append(f"- **近 24 小時**：{lq.get('recent_24h_entries', 0)}\n")
        dashboard.append(f"- **品質分數**：{lq.get('quality_score', 0)}/100\n")
    else:
        dashboard.append("（無日誌品質資料）\n")
    
    dashboard.append("\n---\n\n")
    
    # 趨勢圖（ASCII Art）
    dashboard.append("## 7 天趨勢\n\n")
    if state_health.exists():
        sh = json.loads(state_health.read_text(encoding='utf-8'))
        trend = sh.get('trend', [])
        if trend:
            dashboard.append("### State 目錄大小趨勢\n\n")
            dashboard.append("```\n")
            for entry in trend[-7:]:
                bar_length = int(entry['total_size_mb'] / 0.5)  # 0.5MB = 1 字元
                bar = '█' * min(bar_length, 40)
                dashboard.append(f"{entry['date']}: {bar} {entry['total_size_mb']} MB\n")
            dashboard.append("```\n")
    
    dashboard_text = ''.join(dashboard)
    
    # 寫入檔案
    output_file = PROJECT_ROOT / "docs" / "OPS-DASHBOARD.md"
    output_file.write_text(dashboard_text, encoding='utf-8')
    
    print(f"儀表板已生成：{output_file}")

if __name__ == "__main__":
    generate_dashboard()
